Table dim_customer {
  customer_id varchar
  country varchar
  iso varchar
}

Table dim_product {
  product_id varchar [primary key]
  stock_code varchar
  description varchar
  price float
}

Table dim_date {
  datetime_id varchar [primary key]
  datetime datetime
  year integer
  month integer
  day integer
  hour integer
  minute integer
  weekday integer
}

Table fct_invoices {
  invoice_id string [primary key]
  datetime_id varchar
  product_id varchar
  customer_id varchar
  quantity integer
  total float
}
Ref: fct_invoices.customer_id > dim_customer.customer_id // many-to-one

Ref: fct_invoices.datetime_id > dim_date.datetime_id // many-to-one

Ref: fct_invoices.product_id > dim_product.product_id // many-to-one
